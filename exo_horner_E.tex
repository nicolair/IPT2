\begin{algorithm}
  $A\leftarrow$ liste de $n+1$ nombres indexée à partir de $0$.\;
  \Comment{on note $a_i$ le nombre désigné par $A[i]$.}
  $y\leftarrow A[n]$\;
  $i\leftarrow n$\;
  \Tq{i $\geq$ 1}{
    $i \leftarrow i-1$\;
    $y\leftarrow A[i] + x * y$\;
  }
  \caption{Méthode de Hörner}
  \label{exo_horner_E_1}
\end{algorithm}
